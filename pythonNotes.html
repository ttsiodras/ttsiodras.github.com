Python 2.6 introduces class decorators. This allows us to alter the behaviour of all (or some) methods via a simple '@decorator' syntax. For example, if we want to add logging of all calls made to a class's methods, we simply need this code: 
#!/usr/bin/env python2.6

# Python 2.6 allows for class decorators...
# This allows us to alter the behaviour of methods
# just by prepending a class declaration with a @whatever...

# This is an example "logging" decorator


def DecorateFunction(f):
    def logged(*args, **kwargs):
        print "======="
        print "Method", f.func_name, "was called"
        print "args  :", str(*args[1:])
        print "kwargs:", str(**kwargs)
        print "======="
        f(*args, **kwargs)
    return logged


def DecorateClass(x):
    for i, j in x.__dict__.items():
        if hasattr(j, 'func_code'):
            x.__dict__[i] = DecorateFunction(j)
    return x

# This is the user code...


@DecorateClass
class A:
    def __init__(self):
        self._x = 1

    def Print(self):
        print self._x

    def IncreaseBy(self, y):
        self._x += y

if __name__ == "__main__":
    a = A()
    a.IncreaseBy(5)
    a.Print()
    a.IncreaseBy(11)
    a.Print()
